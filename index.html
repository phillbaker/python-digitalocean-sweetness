#!/usr/bin/python

import digitalocean

manager = digitalocean.Manager(client_id=id, api_key=key)

droplet = digitalocean.Droplet(
	client_id=id,
	api_key=key,
	name = 'do1.phillbaker.com',
	region_id=ny_region_id,
	image_id=ubuntu_1204_image_id,
	size_id=small_size_id,
	backup_active=False
)
droplet.create() # Boom!
